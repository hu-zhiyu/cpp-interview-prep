### 请实现 copyRandomList 函数，复制一个复杂链表。在复杂链表中，每个节点除了有一个 next 指针指向下一个节点，还有一个 random 指针指向链表中的任意节点或者 null。
```c
/*
// Definition for a Node.
class Node {
public:
    int val;
    Node* next;
    Node* random;
    
    Node(int _val) {
        val = _val;
        next = NULL;
        random = NULL;
    }
};
*/
class Solution {
    // 哈希表法, O(n)的时间和空间复杂度
public:
    Node* copyRandomList(Node* head) {
        if(head == nullptr) return nullptr;
        unordered_map<Node*, Node*> nodes;
        Node* pNode = head;
        while(pNode){  // 添加键值对
            nodes[pNode] = new Node(pNode->val);
            pNode = pNode->next;
        }
        pNode = head;
        while(pNode){  // 添加next和random指针
            nodes[pNode]->next = nodes[pNode->next];
            nodes[pNode]->random = nodes[pNode->random];
            pNode = pNode->next;        
        }
        return nodes[head];
    }
};
```
```c
class Solution {
public:
    Node* copyRandomList(Node* head) {
        //方法2: 时间复杂度O(n), 空间复杂度O(1）
        if(head == nullptr) return nullptr;

        //复制节点
        Node *pNode = head;
        while(pNode != nullptr)
        {
            Node *pNext = pNode->next;
            Node *newNode = new Node(pNode->val);
            pNode->next = newNode;
            newNode->next = pNext;
            pNode = pNext;
        }

        //复制随机指针
        Node *preNode = head;
        while(preNode != nullptr)
        {
            pNode = preNode->next;
            pNode->random = preNode->random == nullptr ? nullptr : preNode->random->next;
            preNode = pNode->next;
        }

        //拆分成两个链表
        preNode = head;
        Node *newHead = preNode->next;
        while(preNode != nullptr)
        {
            pNode = preNode->next;

            preNode->next = pNode->next;
            preNode = preNode->next;

            pNode->next =  preNode == nullptr ? nullptr : preNode->next;
        }
        return newHead;
    }
};
```
